# 1) Committi il tuo lavoro
git add .
git commit -m "Messaggio"

# 2) Provi a pushare
git push   # può fallire se il remoto è avanti

# 3) Allinea i riferimenti remoti
git fetch origin

# 4) Rebase dei tuoi commit sopra origin/main
git rebase origin/main

# 5) Se ci sono conflitti:
#    - risolvi i file (<<<<<<< ======= >>>>>>>)
git add .
oppure
git add <file_risolto>  # ripeti per tutti i file risolti

# testa: engine, estensione, script, ecc.
# Se per far passare i test devi modificare altro codice, fallo ORA:
# queste modifiche verranno incluse nel commit in corso di riapplicazione.

# 6) Continua il rebase
git rebase --continue

# (se compaiono altri conflitti, ripeti 5 → 6)

# 7) Rebase finito: push
git push
