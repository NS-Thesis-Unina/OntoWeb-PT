    actor U as Utente
    participant H as Pagina HTTP Requests
    participant S as HttpRequestsService (backend)
    participant G as DataGrid
    participant N as Notifiche globali

    U->>H: Innesca caricamento lista (apertura pagina, Apply o Reset filtri, cambio pagina/pageSize)
    activate H
    H->>S: listHttpRequests(offset, limit, filtri)


    alt Nessun risultato
        activate S
        S-->>H: rows=[] , total=0
        deactivate S
        H->>H: loading=false, aggiorna stato con rows vuote

        activate S
        S-->>H: errore
        deactivate S
        H->>H: loading=false, non altera eventuali rows precedenti
        H->>N: mostra snackbar "Error while executing the request."

    end
